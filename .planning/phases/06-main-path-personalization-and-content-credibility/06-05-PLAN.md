## Goal
Add round 4 (calibration after 5 lessons) and round 5 (interest expansion after 1 track) questions, DTOs, submit endpoints, and profile update logic. Extend VALID_ROUNDS and getStatus to cover rounds 4-5.

## must_haves
- Round 4 questions: career goal confidence, skill self-rating update, learning pace satisfaction
- Round 5 questions: new interest areas, preferred depth, adjust main path suggestion
- POST /onboarding/submit/round-four saves answers and updates LearnerProfile
- POST /onboarding/submit/round-five saves answers and updates LearnerProfile.preferredTopics
- getStatus.completedRounds includes rounds 4-5 when completed
- Submitting a later round removes the corresponding PendingOnboardingRound

2. Add `submitRoundFour` method to `OnboardingService`.
   NOTE: No round-ordering prerequisite — round 4 triggers independently via the 5-lesson milestone (D-05). A user might complete a track (triggering round 5) before completing 5 individual lessons (round 4). Only check for duplicate submission.

```typescript
async submitRoundFour(
  userId: string,
  dto: SubmitRoundFourDto,
): Promise<OnboardingRound> {
  const existingRoundFour = await this.prisma.onboardingRound.findUnique({
    where: { userId_roundNumber: { userId, roundNumber: 4 } },
  });

  if (existingRoundFour) {
    throw new ConflictException('Round 4 already submitted');
  }

  const onboardingRound = await this.prisma.onboardingRound.create({
    data: {
      userId,
      roundNumber: 4,
      answers: {
        careerGoalConfidence: dto.careerGoalConfidence,
        updatedSkillRating: dto.updatedSkillRating,
        learningPaceSatisfaction: dto.learningPaceSatisfaction,
      },
      completedAt: new Date(),
    },
  });

  // Remove pending round record
  await this.prisma.pendingOnboardingRound.deleteMany({
    where: { userId, roundNumber: 4 },
  });

  await this.learnerProfileService.updateFromRoundFour(userId);

  return onboardingRound;
}
```

```typescript
async updateFromRoundFour(userId: string): Promise<void> {
  const round = await this.prisma.onboardingRound.findUnique({
    where: { userId_roundNumber: { userId, roundNumber: 4 } },
  });

  if (!round) {
    this.logger.debug(`No round 4 data for user ${userId}, skipping`);
    return;
  }

  const profile = await this.prisma.learnerProfile.findUnique({
    where: { userId },
  });

  if (!profile) {
    this.logger.debug(`No learner profile for user ${userId}, skipping round 4 update`);
    return;
  }

  const answers = round.answers as Record<string, unknown>;
  const updatedSkillRating = answers.updatedSkillRating as string;
  const learningPaceSatisfaction = answers.learningPaceSatisfaction as string;

  // Update skillLevel from calibrated self-rating
  let skillLevel: LearnerSkillLevel = profile.skillLevel;
  if (updatedSkillRating === 'confident') {
    skillLevel = LearnerSkillLevel.ADVANCED;
  } else if (updatedSkillRating === 'comfortable' || updatedSkillRating === 'improving') {
    skillLevel = LearnerSkillLevel.INTERMEDIATE;
  } else {
    skillLevel = LearnerSkillLevel.BEGINNER;
  }

  // Update learningPace from satisfaction
  let learningPace: LearnerLearningPace = profile.learningPace;
  if (learningPaceSatisfaction === 'too_slow') {
    learningPace = LearnerLearningPace.FAST;
  } else if (learningPaceSatisfaction === 'too_fast') {
    learningPace = LearnerLearningPace.SLOW;
  } else {
    learningPace = LearnerLearningPace.NORMAL;
  }

  await this.prisma.learnerProfile.update({
    where: { userId },
    data: {
      skillLevel,
      learningPace,
      lastRecalculatedAt: new Date(),
    },
  });
}

async updateFromRoundFive(userId: string): Promise<void> {
  const round = await this.prisma.onboardingRound.findUnique({
    where: { userId_roundNumber: { userId, roundNumber: 5 } },
  });

  if (!round) {
    this.logger.debug(`No round 5 data for user ${userId}, skipping`);
    return;
  }

  const profile = await this.prisma.learnerProfile.findUnique({
    where: { userId },
  });

  if (!profile) {
    this.logger.debug(`No learner profile for user ${userId}, skipping round 5 update`);
    return;
  }

  const answers = round.answers as Record<string, unknown>;
  const newInterestAreas = Array.isArray(answers.newInterestAreas)
    ? (answers.newInterestAreas as string[]).filter(a => a !== 'none')
    : [];

  const existingTopics = Array.isArray(profile.preferredTopics)
    ? (profile.preferredTopics as string[])
    : [];

  const preferredTopics = [...new Set([...existingTopics, ...newInterestAreas])];

  await this.prisma.learnerProfile.update({
    where: { userId },
    data: {
      preferredTopics,
      lastRecalculatedAt: new Date(),
    },
  });
}
```
